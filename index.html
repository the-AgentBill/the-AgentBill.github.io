Initialize once, protect any route with requirePayment()

import express from "express";
import { agentBill, requirePayment } from "@agent-bill/sdk";

const app = express();

// 1. Initialize once
agentBill.init({
  receivingAddress: "0xYourAddress",
  network: "base-sepolia",
});

// 2. Protect any route
app.get("/api/weather",
  requirePayment({ amount: "0.01", currency: "USDC" }),
  (req, res) => {
    res.json({ city: "Nairobi", temp: "24°C" });
  }
);

app.listen(3000);
import { Hono } from "hono";
import { agentBill } from "@agent-bill/sdk";
import { requirePayment } from "@agent-bill/sdk/hono";

agentBill.init({
  receivingAddress: "0xYourAddress",
  network: "base-sepolia",
});

const app = new Hono();

app.get("/api/weather",
  requirePayment({ amount: "0.01", currency: "USDC" }),
  (c) => c.json({ city: "Nairobi", temp: "24°C" })
);

export default app;
import Fastify from "fastify";
import { agentBill } from "@agent-bill/sdk";
import { requirePayment } from "@agent-bill/sdk/fastify";

agentBill.init({
  receivingAddress: "0xYourAddress",
  network: "base-sepolia",
});

const app = Fastify();

app.register(requirePayment({ amount: "0.01", currency: "USDC" }), {
  prefix: "/api/weather",
});

app.get("/api/weather", async () => {
  return { city: "Nairobi", temp: "24°C" };
});

app.listen({ port: 3000 });

How it works

1

Client makes a request

An AI agent or API consumer hits your protected endpoint without a payment header.

2

Server responds with 402

AgentBill returns a 402 Payment Required response with payment requirements (amount, address, network).

3

Client signs and retries

The paying client signs a USDC permit and retries the request with a X-PAYMENT header.

4

Payment settles on-chain

AgentBill verifies the signature, runs your handler, then settles the payment via the x402 facilitator. USDC lands in your wallet.
